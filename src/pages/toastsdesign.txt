  <script>
    const toastMessages = {
      processing: {
        icon: '📊',
        title: 'Processing rows',
        description: 'Processing 1,247 rows...',
        class: 'processing'
      },
      duplicates: {
        icon: '🔍',
        title: 'Tagging duplicates', 
        description: 'Identifying and tagging duplicates...',
        class: 'duplicate'
      },
      categorizing: {
        icon: '📂',
        title: 'Categorizing contacts',
        description: 'Categorizing contacts via repository...',
        class: 'categorizing'
      },
      complete: {
        icon: '✅',
        title: 'Processing complete',
        description: 'Download ready.',
        class: 'complete'
      }
    };

    function createToast(type) {
      const container = document.getElementById('toastContainer');
      const toast = document.createElement('div');
      toast.className = `toast ${toastMessages[type].class}`;
      toast.innerHTML = `
        <div class="toast-icon">${toastMessages[type].icon}</div>
        <div class="toast-content">
          <div class="toast-title">${toastMessages[type].title}</div>
          <div class="toast-description">${toastMessages[type].description}</div>
        </div>
      `;
      container.insertBefore(toast, container.firstChild);
      
      // Auto-remove after 5s
      setTimeout(() => {
        if (toast.parentNode) toast.remove();
      }, 5000);
    }

    function showProcessingToast() {
      createToast('processing');
    }

    function showFullSequence() {
      createToast('processing');
      setTimeout(() => createToast('duplicates'), 800);
      setTimeout(() => createToast('categorizing'), 1600);
      setTimeout(() => createToast('complete'), 2400);
    }

    // Connect to process button
    document.addEventListener('DOMContentLoaded', () => {
      const processBtn = document.querySelector('.process-btn');
      const fileInput = document.getElementById('crm-upload');
      processBtn.onclick = () => {
        if (!fileInput.files.length) {
          createToast('complete');
          return;
        }
        showFullSequence();
      };
    });
  </script>
